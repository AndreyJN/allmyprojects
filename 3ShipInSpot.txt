               System.out.println(spot);
               //System.out.println(spotTwo);
               int fire = 0;
                System.out.println("Your Shot");

                   if (scanner.hasNext()) {
                       fire = scanner.nextInt();
                       break;

                       String temp = scanner.nextLine();
                   }
                   }
                
                        switch (spot[fire]) {
                        case '.':
                            System.out.println("Miss");

                            System.out.println("Hit");
                            spot[fire] = '0';
                            break;
